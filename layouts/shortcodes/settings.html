{{/* {{< settings component="runner" >}} {{< settings component="server" exclude="Debugging,DebugDataSeeder,PreviewFeatures" >}} Renders a settings reference for one Snap CD component from the JSON Schema at data/schemas/.json. Each top-level section becomes a collapsible
block; nested objects recurse. Sourced from the same JSON Schema that operators consume via "$schema" in their appsettings.json, so docs and IntelliSense can't drift. Params: component — required; the data file basename under data/schemas/ exclude — optional; comma-separated list of top-level section names to skip */}} {{- $component := .Get "component" -}} {{- if not $component -}} {{- errorf "{{< settings >}} requires a `component` parameter, e.g. {{< settings component=\"runner\" >}}" -}} {{- end -}} {{- $excludeRaw := .Get "exclude" | default "" -}} {{- $exclude := slice -}} {{- range (split $excludeRaw ",") -}} {{- $trimmed := trim . " " -}} {{- if $trimmed -}}{{- $exclude = $exclude | append $trimmed -}}{{- end -}} {{- end -}} {{- $schema := index .Site.Data.schemas $component -}} {{- if not $schema -}}

Settings schema not found for component {{ $component }}. Expected data/schemas/{{ $component }}.json.

{{- else -}}
{{- range $name, $section := $schema.properties -}} {{- if not (in $exclude $name) -}} {{- partial "settings/render.html" (dict "name" $name "schema" $section "topLevel" true) -}} {{- end -}} {{- end -}}
{{- end -}}
